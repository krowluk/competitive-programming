#include <stdio.h>
#include <string.h>

int main(){
 char input[255] = " ";
 int found = 0;

 while(scanf("%s", &input) != EOF){
  scanf("%[^\n]s", &input);
  int len_in = strlen(input);
  for(int i = 0; i < len_in ; i++){
   if(input[i] == '\"'){
    if(found == 0){
     printf("``");
     found = 1;
    }
    else{
     found = 0;
     printf("\'\'");
    }
   }
   else
    printf("%c", input[i]);
  }
  printf("\n");
 }
 return 0;
}
