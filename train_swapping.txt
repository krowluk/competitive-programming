#include <stdio.h>

int main(){
 int cases, number, train[255], count, temp;

 scanf("%d", &cases);

 for(int x = 0; x < cases; x++){
  scanf("%d", &number);
  count = 0;

  for(int y = 0; y < number; y++){
   scanf("%d", &train[y]);
  }

  for(int i = 0; i < number; i++){
   for(int j = 0; j < i; j++){
    if(train[j] > train[i]){
     count ++; 
     temp = train[j];
     train[j] = train[i];
     train[i]  = temp;
    }
   }
  }
  printf("Optimal train swapping takes %d swaps.\n", count);

 }

 return 0;
}
