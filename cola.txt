#include <stdio.h>

int main(){
 int cola, empty, total;
 while(scanf("%d", &cola) != EOF){
  empty = total = cola;
  while(empty >= 3){
   cola = empty / 3;
   empty %= 3;
   total += cola;
   empty += cola;
  }

  if(empty == 2) total++;
  printf("%d\n", total);
 }
}
