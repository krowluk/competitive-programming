#include <stdio.h>

int main(){
 int n, count;
 long num, rev, temp, before;

 scanf("%d", &n);

 for(int i = 0; i < n; i++){
  scanf("%ld", &num);
  count = 0;
  while(true){
   rev = 0;
   before = num;
   while(num != 0){
    temp = num%10;
    rev = rev*10 + temp;
    num /= 10;
   } 
   if(before == rev) break;
   num = before + rev;
   count++;
  }
  printf("%d %ld\n", count, before);
 }
 return 0;
}
